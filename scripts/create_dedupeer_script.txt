create keyspace Dedupeer
    with strategy_options = {replication_factor:1}
    and placement_strategy = 'org.apache.cassandra.locator.SimpleStrategy';

use Dedupeer;
create column family Chunks
    with column_type='Super'
    and comparator = BytesType
    and default_validation_class = 'BytesType'
    and key_validation_class = 'BytesType';
assume Chunks keys as BytesType;

create column family UserFiles 
    with column_type='Super'
    and comparator = UTF8Type
    and default_validation_class = 'UTF8Type'
    and key_validation_class = 'UTF8Type';
assume UserFiles keys as utf8;
assume UserFiles sub_comparator as utf8;

create column family Files
    with comparator = UTF8Type
    and default_validation_class = 'UTF8Type'
    and key_validation_class = 'UTF8Type';
